# サーバー側で実行するコマンド

cd /home/besttrust/kimito-link.com/_git/mailloop

# バックアップ
cp public/index.php public/index.php.bak_$(date +%Y%m%d_%H%M%S) 2>/dev/null || true

# Base64デコードして復元（index_php_base64.txtの内容をそのまま使用）
cat index_php_base64.txt | base64 -d > public/index.php

# 確認
ls -lh public/index.php
head -5 public/index.php

# OPcacheクリア
php public/clear_cache.php 2>&1 || true

echo "復元完了"
